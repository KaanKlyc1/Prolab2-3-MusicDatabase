    private void kayıtTamamlaActionPerformed(java.awt.event.ActionEvent evt) {                                             
        
        String kullanıcıadı= txtf_kullanıcıadı.getText();
        String şifre= pf_şifre.getText();
        String email= txtf_email.getText();
        String abonelik= txtf_abonelik.getText();
        String ülke= txtf_ülke.getText();
        
        String kaydet ="INSERT INTO kullanicilar (kullanici_ad,sifre,email,abonelik,kullanici_ulke) VALUES(?,?,?,?,?)";
        try{
            Veritabanı.pstmt= Veritabanı.con.prepareStatement(kaydet);
            Veritabanı.pstmt.setString(1, kullanıcıadı);
            Veritabanı.pstmt.setString(2, şifre);
            Veritabanı.pstmt.setString(3, email);
            Veritabanı.pstmt.setString(4, abonelik);
            Veritabanı.pstmt.setString(5, ülke);
            Veritabanı.pstmt.executeUpdate();
            System.out.println("Kayıt Tamamlandı.");
        }catch(Exception e){
        System.out.println(e);
        }
        
        Proje3.Giriş_seçimi = new Giriş_seçimi();
        Proje3.Giriş_seçimi.setLocationRelativeTo(null);
        Proje3.Giriş_seçimi.setVisible(true);

    }

    public void güncelle(){
        int i,j;
        try {
            Veritabanı.pstmt= Veritabanı.con.prepareStatement("SELECT * FROM sanatcilar");
            Veritabanı.rs= Veritabanı.pstmt.executeQuery();
            ResultSetMetaData AlbumData = Veritabanı.rs.getMetaData();
            i = AlbumData.getColumnCount();
            DefaultTableModel RecordTable = (DefaultTableModel)sanatçıTablo.getModel();
            RecordTable.setRowCount(0);
            while(Veritabanı.rs.next()){
                Vector columnData = new Vector();
                for(j=1;j<=i;j++){
                    columnData.add(Veritabanı.rs.getString("sanatci_id"));
                    columnData.add(Veritabanı.rs.getString("sanatci_ad"));
                    columnData.add(Veritabanı.rs.getString("sanatci_ulke"));
                }
                RecordTable.addRow(columnData);
            }
        } catch (SQLException ex) {
            Logger.getLogger(Sanatçılar.class.getName()).log(Level.SEVERE, null, ex);
        }
            
         
    }

    public void güncelle(){
        int i,j;
        
        try {
            Veritabanı.pstmt= Veritabanı.con.prepareStatement("SELECT sarkilar_sanatcilar.id,sarkilar.sarki_id,sanatcilar.sanatci_id,sarkilar.sarki_ad,sanatcilar.sanatci_ad  FROM sarkilar JOIN sarkilar_sanatcilar ON (sarkilar.sarki_id=sarkilar_sanatcilar.sarki_id) JOIN sanatcilar ON (sanatcilar.sanatci_id=sarkilar_sanatcilar.sanatci_id)");
            Veritabanı.rs= Veritabanı.pstmt.executeQuery();
            ResultSetMetaData AlbumData = Veritabanı.rs.getMetaData();
            i = AlbumData.getColumnCount();
            DefaultTableModel RecordTable = (DefaultTableModel)şarkısanatçıTablo.getModel();
            RecordTable.setRowCount(0);
            while(Veritabanı.rs.next()){
            Vector columnData = new Vector();
            for(j=1;j<=i;j++){
             columnData.add(Veritabanı.rs.getString("id"));
             columnData.add(Veritabanı.rs.getString("sarki_id"));
             columnData.add(Veritabanı.rs.getString("sanatci_id"));
             columnData.add(Veritabanı.rs.getString("sarki_ad"));
             columnData.add(Veritabanı.rs.getString("sanatci_ad"));
            }
            RecordTable.addRow(columnData);
            }
            
        } catch (SQLException ex) {
            Logger.getLogger(İlişkiler.class.getName()).log(Level.SEVERE, null, ex);
        }
    }

    private void girişButonActionPerformed(java.awt.event.ActionEvent evt) {                                           
        
      
            String kullanıcıadı= txtf_kullanıcıadı.getText();
            String şifre = pf_şifre.getText();
            String girişSorgu = "SELECT kullanici_id,kullanici_ad FROM kullanicilar WHERE kullanici_ad = \'"+kullanıcıadı+"\' and sifre = \'"+şifre+"\'";
            int girilenKullanıcıID;
            String girilenKullanıcıAdı;
            try {
            int giriş=0;
            Veritabanı.stmt= Veritabanı.con.createStatement();
            Veritabanı.rs = Veritabanı.stmt.executeQuery(girişSorgu);
            
            while(Veritabanı.rs.next()){
                giriş=1;
                girilenKullanıcıID = Veritabanı.rs.getInt("kullanici_id");
                girilenKullanıcıAdı= Veritabanı.rs.getString("kullanici_ad");
                Proje3.girilenKullanıcıID= girilenKullanıcıID;
                Proje3.girilenKullanıcıAdı= girilenKullanıcıAdı;
                Proje3.KullanıcıPanel = new KullanıcıPanel();
                Proje3.KullanıcıPanel.setLocationRelativeTo(null);
                Proje3.KullanıcıPanel.setVisible(true);
                this.dispose();
                
            }
            if(giriş==0){
            girişUyarı.setText("Kullanıcı adı veya şifre yanlış!");
            }
        } catch (SQLException ex) {
            Logger.getLogger(GirişEkranı1.class.getName()).log(Level.SEVERE, null, ex);
        }
        
        
    }

        int i,j;
        try {
            Veritabanı.pstmt= Veritabanı.con.prepareStatement("SELECT sarkilar.sarki_id,sarkilar.sarki_ad,sanatcilar.sanatci_ad,albumler.album_ad,sarkilar.sarki_tarih,sarkilar.sarki_tur,sarkilar.sure,sarkilar.dinlenme FROM sarkilar JOIN sarkilar_sanatcilar ON (sarkilar.sarki_id=sarkilar_sanatcilar.sarki_id) JOIN sanatcilar ON (sanatcilar.sanatci_id=sarkilar_sanatcilar.sanatci_id)  JOIN sanatcilar_albumler ON (sanatcilar.sanatci_id=sanatcilar_albumler.sanatci_id) JOIN albumler ON (sanatcilar_albumler.album_id=albumler.album_id) WHERE (sarkilar.album_id=albumler.album_id)");
            Veritabanı.rs= Veritabanı.pstmt.executeQuery();
            ResultSetMetaData ŞarkıData = Veritabanı.rs.getMetaData();
            i = ŞarkıData.getColumnCount();
            DefaultTableModel RecordTable = (DefaultTableModel)şarkıTablo.getModel();
            RecordTable.setRowCount(0);
            while(Veritabanı.rs.next()){
                Vector columnData = new Vector();
                for(j=1;j<=i;j++){
                    columnData.add(Veritabanı.rs.getString("sarki_id"));
                    columnData.add(Veritabanı.rs.getString("sarki_ad"));
                    columnData.add(Veritabanı.rs.getString("sanatci_ad"));
                    columnData.add(Veritabanı.rs.getString("album_ad"));

                    columnData.add(Veritabanı.rs.getString("sarki_tur"));
                    columnData.add(Veritabanı.rs.getString("sure"));
                    columnData.add(Veritabanı.rs.getString("dinlenme"));
                }
                RecordTable.addRow(columnData);
            }

    public void yazdır() {
        try {
            String bilgial = "SELECT kullanici_id,kullanici_ad,email,abonelik,kullanici_ulke FROM kullanicilar WHERE kullanici_id = \'" + Proje3.profilID + "\'";

        }

    }

    public void güncelle() {

        int i, j;
        try {
            Veritabanı.pstmt = Veritabanı.con.prepareStatement("SELECT sarkilar.sarki_id,sarkilar.sarki_ad,sanatcilar.sanatci_ad,albumler.album_ad,sarkilar.sarki_tarih,sarkilar.sure,kullanicilar.kullanici_id FROM sarkilar JOIN sarkilar_sanatcilar ON (sarkilar.sarki_id=sarkilar_sanatcilar.sarki_id) JOIN sanatcilar ON (sanatcilar.sanatci_id=sarkilar_sanatcilar.sanatci_id)  JOIN sanatcilar_albumler ON (sanatcilar.sanatci_id=sanatcilar_albumler.sanatci_id) JOIN albumler ON (sanatcilar_albumler.album_id=albumler.album_id) JOIN sarkilar_kullanicilar ON (sarkilar.sarki_id = sarkilar_kullanicilar.sarki_id) JOIN kullanicilar ON (sarkilar_kullanicilar.kullanici_id=kullanicilar.kullanici_id) WHERE (sarkilar.album_id=albumler.album_id) and (sarkilar.sarki_tur= 'Pop') and (kullanicilar.kullanici_id= \'" + Proje3.profilID + "\')");
            Veritabanı.rs = Veritabanı.pstmt.executeQuery();
            ResultSetMetaData ŞarkıData = Veritabanı.rs.getMetaData();
            i = ŞarkıData.getColumnCount();
            DefaultTableModel RecordTable = (DefaultTableModel) popTablo.getModel();
            RecordTable.setRowCount(0);
            while (Veritabanı.rs.next()) {
                Vector columnData = new Vector();
                for (j = 1; j <= i; j++) {
                    columnData.add(Veritabanı.rs.getString("sarki_id"));
                    columnData.add(Veritabanı.rs.getString("sarki_ad"));
                    columnData.add(Veritabanı.rs.getString("sanatci_ad"));
                    columnData.add(Veritabanı.rs.getString("album_ad"));
                    columnData.add(Veritabanı.rs.getString("sarki_tarih"));
                    columnData.add(Veritabanı.rs.getString("sure"));
                }
                RecordTable.addRow(columnData);
            }
        } catch (SQLException ex) {
            Logger.getLogger(Profil.class.getName()).log(Level.SEVERE, null, ex);
        }

    }
